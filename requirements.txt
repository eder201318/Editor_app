# Usa uma imagem oficial do Python como base
FROM python:3.11-slim

# Instala o FFmpeg e dependências do sistema
RUN apt-get update && apt-get install -y \
    ffmpeg \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Define o diretório de trabalho dentro do servidor
WORKDIR /app

# Copia os ficheiros de requisitos e instala as bibliotecas Python
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copia o resto dos ficheiros do projeto (app.py e index.html)
COPY . .

# Cria as pastas de upload e processamento
RUN mkdir -p uploads processed

# Define a porta que o Render vai usar
ENV PORT=10000

# Comando para iniciar o servidor usando Gunicorn (mais estável para produção)
CMD gunicorn --bind 0.0.0.0:$PORT --timeout 600 app:app
